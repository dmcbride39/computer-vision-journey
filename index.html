import streamlit as st import cv2 import numpy as np from PIL import Image st.title("🖼️ Devin’s Image Processor") st.write("Upload a photo and try the effects below") # File uploader uploaded_file = st.file_uploader("Choose an image file", type=["jpg", "jpeg", "png"]) if uploaded_file is not None: # Load image image = Image.open(uploaded_file) img_array = np.array(image) # Convert to OpenCV format if len(img_array.shape) == 3: original = cv2.cvtColor(img_array, cv2.COLOR_RGB2BGR) else: original = img_array.copy() st.image(image, caption="Original Image", use_column_width=True) st.subheader("Effects") col1, col2, col3, col4 = st.columns(4) with col1: if st.button("🔲 Black & White"): gray = cv2.cvtColor(original, cv2.COLOR_BGR2GRAY) st.image(gray, caption="Black & White", use_column_width=True, channels="GRAY") with col2: if st.button("🌫️ Blur"): blurred = cv2.GaussianBlur(original, (25, 25), 0) st.image(cv2.cvtColor(blurred, cv2.COLOR_BGR2RGB), caption="Blurred Image", use_column_width=True) with col3: if st.button("🔍 Edges"): gray = cv2.cvtColor(original, cv2.COLOR_BGR2GRAY) edges = cv2.Canny(gray, 50, 150) st.image(edges, caption="Edge Detection", use_column_width=True, channels="GRAY") with col4: if st.button("✨ Sharpen"): kernel = np.array([[-1, -1, -1], [-1, 9, -1], [-1, -1, -1]]) sharpened = cv2.filter2D(original, -1, kernel) st.image(cv2.cvtColor(sharpened, cv2.COLOR_BGR2RGB), caption="Sharpened", use_column_width=True) # Extra feature if st.button("📏 Resize to Small"): small = cv2.resize(original, (500, 400)) st.image(cv2.cvtColor(small, cv2.COLOR_BGR2RGB), caption="Resized Image", use_column_width=True)
